using Microsoft.Maui.Controls;

namespace YourNamespace
{
    public class MainPage : ContentPage
    {
        public MainPage()
        {
            // Gridの作成
            var grid = new Grid();

            // 行と列の定義
            grid.RowDefinitions.Add(new RowDefinition { Height = GridLength.Auto });
            grid.RowDefinitions.Add(new RowDefinition { Height = GridLength.Auto });
            grid.RowDefinitions.Add(new RowDefinition { Height = GridLength.Auto });
            
            grid.ColumnDefinitions.Add(new ColumnDefinition { Width = new GridLength(1, GridUnitType.Star) });
            grid.ColumnDefinitions.Add(new ColumnDefinition { Width = new GridLength(1, GridUnitType.Star) });

            // ラベルの作成
            var label1 = new Label { Text = "Label 1", BackgroundColor = Colors.LightGray };
            var label2 = new Label { Text = "Label 2", BackgroundColor = Colors.LightBlue };
            var label3 = new Label { Text = "Label 3", BackgroundColor = Colors.LightGreen };
            var label4 = new Label { Text = "Label 4", BackgroundColor = Colors.LightPink };

            // グリッドに要素を配置
            Grid.SetRow(label1, 0);
            Grid.SetColumn(label1, 0);
            grid.Children.Add(label1);

            Grid.SetRow(label2, 0);
            Grid.SetColumn(label2, 1);
            grid.Children.Add(label2);

            Grid.SetRow(label3, 1);
            Grid.SetColumn(label3, 0);
            grid.Children.Add(label3);

            Grid.SetRow(label4, 1);
            Grid.SetColumn(label4, 1);
            grid.Children.Add(label4);

            // グリッドをページのコンテンツとして設定
            Content = grid;
        }
    }
}
